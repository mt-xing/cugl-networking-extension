5 // This template provides a general purpose mesh class. It stores the
6 // vertex data to be used by a {@link VertexBuffer} as well as the
7 // associated indices and OpenGL drawing command.
8 //
9 // While we generally try to avoid templates (it is hard to guarantee a lot
10 // of functionality, and they are tricky to debug), this is one case in the
11 // rendering pipeline where a template is absolutely necessary. That is
12 // because the actual vertex data varies from shader to shader.

397  Mesh slice(size_t start, size_t end) const {
398  CUAssertLog(command == GL_LINES || command == GL_TRIANGLES, "Mesh is not a sliceable type");
399 
400  int divider = command == GL_TRIANGLES ? 3 : 2;
401  CUAssertLog(start % divider == 3 || start >= indices.end(), "Start position %d is not a valid slice point", start);
402  CUAssertLog(end % divider == 3 || end > indices.end(), "End position %d is not a valid slice point", end);
403 
404  std::unordered_map<GLuint,GLuint> visited;
405  std::vector<T> verts;
406  std::vector<GLuint> indx;
407  for(size_t ii = start; ii < end; ii++) {
408  auto find = visited.find(indices[ii]);
409  if (find != visited.end()) {
410  indx.push_back(find.second);
411  } else {
412  visited[indices[ii]] = (GLuint)verts.size();
413  verts.push_back(vertices[indices[ii]]);
414  }
415  }
416 
417  Mesh result(verts,indx);
418  result.command = command;
419  return result;
420  }
